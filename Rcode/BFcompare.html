try comparing BF for normal vs servin and stephens bf

lABF = function(k, T) {
    return(((k/2) * T^2))
}
lBF = function(k, T, n) {
    return((-n/2) * log(1 - (k/n) * T^2))
}

n = seq(1, 100, length = 100)
T = 2
k = 0.4/T^2

plot(n, lBF(k, T, n) - lABF(k, T), type = "l")
abline(h = 0, col = 2)
